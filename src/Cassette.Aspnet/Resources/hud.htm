                        To enable Cassette tracing please add this to web.config:
                    

                    
<system.diagnostics>
    <sources>
        <source name="Cassette" switchName="CassetteSwitch" switchType="System.Diagnostics.SourceSwitch" />
    </sources>
    <switches>
        <add name="CassetteSwitch" value="All" />
    </switches>
</system.diagnostics>
